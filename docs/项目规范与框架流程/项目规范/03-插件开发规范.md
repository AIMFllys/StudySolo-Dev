## 2. 节点插件

### 2.1 节点分类

| 类型 | 说明 | 复杂度 | 开发文件数 |
|------|------|--------|----------|
| A 型 | 纯 LLM Prompt 节点 | 低 | 2-3 个文件 |
| B 型 | 外部工具节点 | 中 | 3-5 个文件 |
| C 型 | 全功能插件 | 高 | 10-30 个文件 |

### 2.2 节点开发流程

**Step 0：分类判断**
- 是否需要外部 API 调用？
- 是否需要新增数据库表？
- 是否需要新的前端页面？

**Step 1：创建节点目录**

```bash
# A/B 型节点
backend/app/nodes/nodes/<category>/<node_type>/
  ├── __init__.py
  ├── node.py
  └── prompt.md  # A 型需要

# C 型或社区插件
backend/app/nodes/plugins/<plugin_name>/
  ├── __init__.py
  ├── node.py
  ├── service.py  # B 型需要
  ├── prompt.md
  ├── plugin.json
  └── README.md
```

**Step 2：实现节点类**

```python
# backend/app/nodes/nodes/generation/quiz_gen/node.py
from app.nodes._base import BaseNode, NodeInput

class QuizGenNode(BaseNode):
    node_type = "quiz_gen"
    category = "generation"
    description = "生成测验题"
    is_llm_node = True
    output_format = "json"

    async def execute(self, node_input: NodeInput, llm_caller):
        messages = self.build_messages(node_input)
        async for token in llm_caller(messages):
            yield token

    async def post_process(self, raw_output: str):
        # 验证 JSON 输出
        data = json.loads(raw_output)
        return NodeOutput(content=json.dumps(data, ensure_ascii=False), format="json")
```

**Step 3：前端渲染器**

```typescript
// frontend/src/features/workflow/components/nodes/renderers/QuizRenderer.tsx
export function QuizRenderer({ data }: { data: QuizNodeData }) {
  const quiz = JSON.parse(data.output || '{}');
  return (
    <div>
      {quiz.questions?.map((q, i) => (
        <div key={i}>{q.title}</div>
      ))}
    </div>
  );
}
```

**Step 4：注册**

```typescript
// frontend/src/features/workflow/components/nodes/renderers/index.ts
import { QuizRenderer } from './QuizRenderer';

export const RENDERER_REGISTRY = {
  quiz_gen: QuizRenderer,
  // ...
};
```

## 3. 前端插件

### 3.1 插件目录结构

```
frontend/src/plugins/
├── registry.ts              # 插件注册表
├── README.md
└── my-custom-plugin/        # 插件示例
    ├── plugin.json
    ├── renderer.tsx
    └── node-config.ts
```

### 3.2 插件注册

```typescript
// frontend/src/plugins/registry.ts
interface Plugin {
  id: string;
  name: string;
  nodeTypes: string[];
  renderer: React.ComponentType;
}

const plugins: Plugin[] = [];

export function registerPlugin(plugin: Plugin) {
  plugins.push(plugin);
  RENDERER_REGISTRY[plugin.nodeTypes[0]] = plugin.renderer;
}

export function getPlugins(): Plugin[] {
  return [...plugins];
}
```

## 4. 子后端 Agent

### 4.1 Agent 目录结构

```
services/
├── code-review-agent/           # 示例 Agent
│   ├── src/
│   │   ├── __init__.py
│   │   ├── main.py            # FastAPI 入口
│   │   ├── agent.py          # Agent 逻辑
│   │   ├── prompts.py         # Prompt 模板

│   ├── .env.example

└── custom-agent-template/       # 模板

    └── README.md
```

### 4.3 API Key 管理

每个 Agent 的 API Key 存储在：
1. Agent 自己的 `.env` 文件（不在主仓库）
2. 主后端只存储调用地址

```yaml
# 主后端 config.yaml
subagents:
  code_review:
    url: ${CODE_REVIEW_AGENT_URL}
```

## 5. 发布流程

### 5.1 节点发布

1. 完善 `plugin.json`
2. 创建 PR 到主仓库
3. Code review 通过后合并

### 5.2 Agent 发布

1. 完善 Dockerfile 和 README
2. 配置 CI/CD（GitHub Actions）
3. 发布 Docker 镜像
4. 更新主后端配置
