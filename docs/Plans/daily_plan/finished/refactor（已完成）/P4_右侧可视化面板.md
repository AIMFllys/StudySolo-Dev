function ThinkingEntry({ entry }: { entry: ThinkingEntry }) {
  const icons: Record<string, string> = {

    zone_start: getZoneIcon(entry.zoneId),

  };

          ✓ 完成
        </div>
      )}
    </div>
  );
}

#### T4B.3: AIProcessTab 组件

**新建文件**: `frontend/src/features/workflow/components/panel/tabs/AIProcessTab.tsx`

```typescript
function AIProcessTab() {
  const { entries, isActive } = useAIProcessStore();
  const scrollRef = useRef<HTMLDivElement>(null);
  
  // 新条目出现时自动滚动到底部
  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTo({
        top: scrollRef.current.scrollHeight,
        behavior: 'smooth',
      });
    }
  }, [entries.length]);
  
  if (entries.length === 0) {
    return (
      <EmptyState
        icon="neurology"
        title="AI 过程"
        description="当 AI 生成工作流时，思考过程将实时展示在这里。"
      />
    );
  }
  
  return (
    <div ref={scrollRef} className="flex-1 overflow-y-auto px-1">
      {/* 时间轴布局 */}
      <div className="space-y-0">
        {entries.map((entry) => (
          <ThinkingEntry key={entry.id} entry={entry} />
        ))}
      </div>
      
      {isActive && (
        <div className="flex items-center gap-2 py-3 pl-4 text-xs text-sky-300/70">
          <span className="inline-block h-2 w-2 animate-pulse rounded-full bg-sky-400" />
          AI 正在工作...
        </div>
      )}
    </div>
  );
}
```

#### T4B.4: TypewriterText 组件

**新建文件**: `frontend/src/components/ui/TypewriterText.tsx`

```typescript
function TypewriterText({ text, speed = 20 }: { text: string; speed?: number }) {
  const [displayedLength, setDisplayedLength] = useState(0);
  
  useEffect(() => {
    if (displayedLength >= text.length) return;
    
    const timer = setTimeout(() => {
      setDisplayedLength(prev => Math.min(prev + 1, text.length));
    }, speed);
    
    return () => clearTimeout(timer);
  }, [displayedLength, text, speed]);
  
  return (
    <span>
      {text.slice(0, displayedLength)}
      {displayedLength < text.length && (
        <span className="animate-blink text-primary">▍</span>
      )}
    </span>
  );
}
```

### Phase 4C: 面板交互优化 (2-3h)

#### T4C.1: 区域可折叠

每个区域 (状态统计 / 当前焦点 / 节点配置 / 节点列表) 支持点击标题折叠/展开。

```typescript
function CollapsibleSection({ title, defaultOpen = true, children }) {
  const [open, setOpen] = useState(defaultOpen);
  return (
    <section>
      <button onClick={() => setOpen(!open)} className="...">
        <span className={`transition-transform ${open ? 'rotate-0' : '-rotate-90'}`}>
          ▼
        </span>
        {title}
      </button>
      {open && children}
    </section>
  );
}
```

#### T4C.2: 节点列表优化

- 点击节点列表项 → 画布自动居中到对应节点 (`fitView` with `nodes: [clickedNode]`)
- 执行中的节点显示实时进度 (闪烁效果)
- 已完成的节点显示输出内容预览 (可展开)

#### T4C.3: 自动切换 Tab

```
事件: AI 开始生成工作流 → 自动切换到 "🧠 AI 过程" Tab

事件: 用户点击画布节点 → 自动切换到 "📊 工作流总览" Tab
```
